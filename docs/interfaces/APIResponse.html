Interface APIResponse<T>

Standard API response wrapper.

Remarks

All WIIL Platform API responses follow this structure, wrapping the actual data with metadata about the request.

Example

const response: APIResponse<Organization> = {
  success: true,
  data: {
    id: 'org_123',
    companyName: 'Acme Corp',
    // ...
  },
  metadata: {
    timestamp: 1704067200000,
    version: 'v1'
  }
};
interface APIResponse<T> {
    success: boolean;
    data: T;
    metadata: { timestamp: number; version: string };
}
